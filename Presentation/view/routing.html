  • Create a new scope

Sample Code

	
angular.module('myApp', []).
config(['$routeProvider', function($routeProvider) {
	$routeProvider
	.when('/', {
	templateUrl: 'views/home.html',
	controller: 'HomeController'
	})
	.when('/login', {
	templateUrl: 'views/login.html',
	controller: 'LoginController'
	})
	.otherwise({
	redirectTo: '/'
	});
}]);

UI-Router

  • The UI-Router is a routing framework for AngularJS 
  • It provides a different approach than ngRoute in that it changes application views based on state of the application and not just the route URL.
  • With this views and routes aren’t tied down to the site URL. 
  •  We use 
<div ui-view></div> instead of ngRoute’s <div ng-view></div>.

EXAMPLE

//HTML
 <div class="container">
  <!-- THIS IS WHERE WE WILL INJECT OUR CONTENT -->
    <div ui-view></div>
</div>

//js
var routerApp = angular.module('routerApp', ['ui.router']);
routerApp.config(function($stateProvider, $urlRouterProvider) {    
    $urlRouterProvider.otherwise('/home');    
    $stateProvider        
        // HOME STATES AND NESTED VIEWS 
        .state('home', {
            url: '/home',
            templateUrl: 'partial-home.html'
        })        
        // ABOUT PAGE AND MULTIPLE NAMED VIEWS 
        .state('about', {
            // we'll get to this in a bit       
        });        
});
